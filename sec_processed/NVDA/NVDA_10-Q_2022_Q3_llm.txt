def extract_revenue(document, context_code):
    # Find the Revenue section
    revenue_section = find_section(document, '<Revenue>')
    
    # Extract all line items with their context codes
    revenue_data = {}
    totals = {}
    parent_child_map = {}
    
    # First pass: collect all items and identify parent-child relationships
    for line in revenue_section:
        if context_code in line.split('|')[1].strip():
            columns = line.split('|')
            raw_item_name = columns[0].strip()
            
            # Parse hierarchy information
            if raw_item_name.startswith('@TOTAL:'):
                # This is a total item
                item_name = raw_item_name.replace('@TOTAL:', '').strip()
                is_total = True
                level = 0  # Top level
            elif raw_item_name.startswith('@SUBTOTAL:'):
                # This is a subtotal item
                item_name = raw_item_name.replace('@SUBTOTAL:', '').strip()
                is_total = True
                level = 1  # Mid level
            elif '@CHILD_OF:' in raw_item_name:
                # This is a child item, extract its parent
                parts = raw_item_name.split('@CHILD_OF:')
                item_name = parts[0].replace('↳', '').strip()
                parent_name = parts[1].strip()
                parent_child_map[item_name] = parent_name
                is_total = False
                level = 2  # Child level
            else:
                # Regular item
                item_name = raw_item_name.strip()
                is_total = False
                level = 1  # Base level
            
            # Find the value corresponding to the context code
            value = get_value_for_context(columns, context_code)
            
            # Store with hierarchy information
            if is_total:
                totals[item_name] = {
                    'value': value,
                    'level': level,
                    'components': []  # Will be filled in second pass
                }
            else:
                revenue_data[item_name] = {
                    'value': value,
                    'level': level,
                    'parent': parent_child_map.get(item_name)
                }
    
    # Second pass: associate children with their parent totals
    for item_name, item_data in revenue_data.items():
        parent = item_data.get('parent')
        if parent and parent in totals:
            totals[parent]['components'].append(item_name)
    
    # Combine the data structures
    result = {
        'line_items': revenue_data,
        'totals': totals,
        # Include verification formulas
        'verify': {
            'sum_components': 'Sum of all components should equal the total',
            'formula': 'Total_Revenue = Product_Revenue + Service_Revenue + Other_Revenue'
        }
    }
    return result
